\section{Weitwurf im See}
\begin{enumerate}[label=\alph*)]
    \item $x = \frac{\lambda \Delta t}{T}$, wobei $T$: Periodendauer, $\Delta t$: Zeit von Eintauchen bis Auftreffen der Wellen aufs Ufer. 
    \item Da die Energie sich kreisförmig ausbreitet, nimmt sie mit dem Abstand schneller ab als bei einer eindimensionalen Ausbreitung.
\end{enumerate}

\section{Seilwelle}
$F = \mu g x  \implies \Delta t = \int_{0}^{L} \frac{dx}{v(x)} = \int_{0}^{L} \frac{dx}{\sqrt{\frac{F(x)}{\mu}}} = \int_{0}^{L} \frac{dx}{\sqrt{\frac{\mu g x}{\mu}}} = \int_{0}^{L} \frac{dx}{\sqrt{gx}} = \dul{2\sqrt{\frac{L}{g}}}$

\section{Transversal gekoppelte Massepunkte}
\begin{enumerate}[label=\alph*)]
    \item $F_{m_1} = m \ddot{y_1} = -ky_1 + l(y_2 - y_1) \implies y_1 = \frac{ly_2  - m \ddot{y_1}}{k + l}$, \\
    $F_{m_2} = m \ddot{y_2} = -ky_2 + l(y_1 - y_2) \implies y_2 = \frac{ly_1 - m \ddot{y_2}}{k + l}$ \\
    
    \item $\delta := y_2 - y_1 = \frac{ly_1 - m \ddot{y_2} - ly_2 + m \ddot{y_1}}{k + l}
    = \frac{- m \ddot{\delta} - l\delta}{k + l} 
    \implies \ddot{\delta} = - \delta \frac{k + 2l}{m}$ \\ 

    Ansatz: $\delta = A_{\delta} \cos(\omega_{\delta} t + \Delta_{\delta})$. 
    Es folgt $\omega_{\delta} = \sqrt{\frac{k + 2l}{m}}$. 
    Die Anfangsbedingungen implizieren $\Delta_{\delta} = 0, A_{\delta} = -y_1^0$.
    
    \item $\sigma := y_1 + y_2 = \frac{ly_2  - m \ddot{y_1} + ly_1 - m \ddot{y_2}}{k + l}
    = \frac{- m\ddot{\sigma} + l \sigma}{k + l} \implies \ddot{\sigma} = -\sigma \frac{k}{m}$ \\

    Ansatz: $\sigma = A_{\sigma} \cos(\omega_{\sigma} t + \Delta_{\sigma})$. 
    Es folgt $\omega_{\sigma} = \sqrt{\frac{k}{m}}$. 
    Die Anfangsbedingungen implizieren $\Delta_{\sigma} = 0, A_{\sigma} = y_1^0$.
    
    \item $y_1(t) = \frac12 (\sigma - \delta) 
    = \frac12 \left(y_1^0 \cos\left(\sqrt{\frac{k}{m}} t\right) + y_1^0 \cos\left(\sqrt{\frac{k + 2l}{m}} t\right)\right)$, \\
    $y_2(t) = \frac12 (\sigma + \delta)
    = \frac12 \left(y_1^0 \cos\left(\sqrt{\frac{k}{m}} t\right) - y_1^0 \cos\left(\sqrt{\frac{k + 2l}{m}} t\right)\right)$ \\
    \vspace{0.5em}
    Die Schwingung "wandert" zwischen den beiden Massen hin und her.

    \item Die Frequenz der Schwebungen ist $\Omega = \left|\omega_{\sigma} - \omega_{\delta}\right| = \left|\sqrt{\frac{k}{m}} - \sqrt{\frac{k + 2l}{m}}\right|$. Daher wird die Energie nach $t = \frac{\pi}{\Omega}$ übertragen. Eine vollständige Übertragung ist jedoch nicht garantiert, da dafür $y_1 = \dot{y_1} = 0$ gelten muss. 
\end{enumerate}
