\section{Geladener Kreisring}
\begin{enumerate}[label=\alph*)]
    \item $\Phi(z) = \int_{a}^{R} \frac{2\pi r \rho \, dr}{4 \pi \epsilon_0 \sqrt{r^2 + z^2} A_{tot}} 
    = \frac{1}{2 \epsilon_0 A_{tot}} \left[\sqrt{r^2 + z^2} \right]_{a}^{R}
    = \dul{\frac{Q \left(\sqrt{R^2 + z^2} - \sqrt{a^2 + z^2}\right)}{4 \pi \epsilon_0 (R^2 - a^2)}}$

    \item $E(z) = -\nabla \Phi(z) 
    = \dul{\frac{Q}{4 \pi \epsilon_0 (R^2 - a^2)} \left(\frac{z}{\sqrt{a^2 + z^2}} - \frac{z}{\sqrt{R^2 + z^2}} \right)}$
    
    \item $E_P(r) = \frac{-Q}{8 \pi \epsilon_0 r^2} \implies E(z) = \dul{\frac{-Q}{8 \pi \epsilon_0 (z \pm b)^2}}$
    
    \item $\Phi_P(z) = \frac{-Q}{8 \pi \epsilon_0 \abs{z \pm b}}$
    
    \item $\Phi(\qty{0}{\meter}) = \frac{Q \left(\sqrt{R^2} - \sqrt{a^2}\right)}{4 \pi \epsilon_0 (R^2 - a^2)} - \frac{Q}{4 \pi \epsilon_0 b}
    = \frac{Q}{4 \pi \epsilon_0} \left(\frac{1}{R + a} - \frac{1}{b}\right) \stackrel{R = 2a}{=} \frac{Q}{4 \pi \epsilon_0} \left(\frac{1}{3a} - \frac{1}{b}\right) = \qty{0}{\volt}
    \implies a = \dul{\frac{b}{3}}$ 
\end{enumerate}

\section{Wasserstoffatom}
\begin{enumerate}[label=\alph*)]
    \item $\int_{0}^{\infty} - 4 \pi r^2 \rho_0 \exp \left(\frac{- 2r}{a}\right) \, dr
    = -4 \pi \rho_0 \left[r^2 \frac{-a \exp \left(\frac{-2r}{a} \right)}{2} 
    - 2r \frac{a^2 \exp \left(\frac{-2r}{a} \right)}{4} 
    + 2 \frac{-a^3 \exp \left(\frac{-2r}{a} \right)}{8} \right]_{0}^{\infty}$ \\
    $ = -\pi \rho_0 a^3 = -e \implies \rho_0 = \frac{e}{\pi a^3} \approx \dul{\qty{3.44e11}{\coulomb \per \cubic \meter}}$

    % 1.602176634e-19 / (pi * (5.29177210903e-11)^3) =344157407973.0031
    
    \item Aufgrund der Symmetrie muss das E-Feld auch symmetrisch sein, das heisst nur von $r$ abhängen. \\
    $Q = e + \int_{0}^{r} - 4 \pi r'^2 \rho_0 \exp \left(\frac{- 2r'}{a}\right) \, dr' 
    = e - 4 \pi \rho_0 \left[r^2 \frac{-a \exp \left(\frac{-2r'}{a} \right)}{2} 
    - 2r' \frac{a^2 \exp \left(\frac{-2r'}{a} \right)}{4} 
    + 2 \frac{-a^3 \exp \left(\frac{-2r'}{a} \right)}{8} \right]_{0}^{r}$ \\
    $ = e - \frac{e \exp \left(\frac{-2r}{a}\right)}{4 \pi \epsilon_0 r^2} \left(1 + \frac{2r}{a} + \frac{2r^2}{a^2} \right) 
    \implies k = \dul{\frac{1}{4 \pi \epsilon_0}}$

\end{enumerate}
