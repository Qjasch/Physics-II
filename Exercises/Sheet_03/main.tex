\section{Dopplereffekt bei Schallwellen}
$\nu' = \frac{\nu c}{c - v} = \frac{\nu c }{c - y_0 \Omega \cos \Omega t} 
\approx \nu \left(1 + \frac{y_0 \Omega \cos \Omega t}{c}\right)$. \\
Die Phase berechnet sich aus dem moduliertem Weg als $\nu \frac{z - y_0 \sin \Omega t}{c}$. \\
Daher gilt insgesamt $f_z(t) = \dul{A \sin \left[2 \pi \nu \left(t + \frac{z - y_0 \Omega \cos \Omega t}{c}\right)\right]}$

    % 1 / (4 * pi * 8.8541878128e-12) * 1.602176634e-19 * ()

    \item $F_A = \frac{1}{4 \pi \epsilon_0} \frac{\overline{q}^2 - \Delta q^2}{r^2}, 
    F_B = \frac{1}{4 \pi \epsilon_0} \frac{\overline{q}^2}{r^2} 
    \implies F_B - F_A = \frac{1}{4 \pi \epsilon_0} \frac{\Delta q^2}{r^2}
    \implies \abs{\Delta q} = r \sqrt{(F_B - F_A) 4 \pi \epsilon_0}$, \\
    $\abs{\overline{q}} = r \sqrt{F_B \cdot 4 \pi \epsilon_0}
    \implies q_1 = r \left[\sqrt{F_B \cdot 4 \pi \epsilon_0} \pm \sqrt{(F_B - F_A) 4\pi \epsilon_0}\right]
    \approx \dul{\qty{-1.2e-15}{\coulomb} \vee \qty{2e-15}{\coulomb}}$, 

    $q_2 = r \left[\sqrt{F_B \cdot 4 \pi \epsilon_0} \mp \sqrt{(F_B - F_A) \pi \epsilon_0} \right]
    \approx \dul{\qty{2e-15}{\coulomb} \vee \qty{-1.2e-15}{\coulomb}}$

\section{Schwingung eines geladenen Teilchens}
\begin{enumerate}[label=\alph*)]
    \item $E(x) = \int_{0}^{2 \pi} \frac{qx \, d\varphi }{4 \pi \epsilon_0 2 \pi (r + x)^2}
    = \frac{qx}{4 \pi \epsilon_0 (r + x)^2} \approx \dul{\frac{qx}{4 \pi \epsilon_0 r^2}}$
    
    \item $F = -q E(x) = \dul{-\frac{q^2 x}{4 \pi \epsilon_0 r^2}}$
    
    \item $F \propto -x \implies$ harmonische Schwinung. 
    
    \item $\omega^2 = \frac{q^2}{4 \pi \epsilon_0 r^2 m} \implies f = \frac{\omega}{2 \pi} = \dul{\frac{q}{2 \pi r} \sqrt{\frac{1}{4 \pi \epsilon_0 m}}}$
\end{enumerate}
