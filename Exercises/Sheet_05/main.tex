\section{Verbindung von Kondensatoren}
\begin{enumerate}[label=\alph*)]
    \item $C_{Gesamt} = C_1 + C_2$, $Q_{Gesamt} = Q_1' + Q_2' = C_1 U 
    \implies U' = \frac{Q_{Gesamt}}{C_{Gesamt}} 
    = \frac{C_1 U}{C_1 + C_2}$. \\
    Damit ergibt sich $Q_1' = C_1 U' = \frac{C_1^2 U}{C_1 + C_2} \approx \dul{\qty{5e-9}{\coulomb}}$ 
    und $Q_2' = C_2 U' = \frac{C_1 C_2 U}{C_1 + C_2} \approx \dul{\qty{4.5e-8}{\coulomb}}$.
    
    \item $E_1 = \frac{C_1 U^2}{2} \approx \dul{\qty{1.25e-4}{\joule}}$,
    $E_{Gesamt}' = \frac{C_{Gesamt} U'^2}{2}
    = \frac{C_1^2 U^2}{2(C_1 + C_2)} \approx \dul{\qty{1.25e-5}{\joule}}$
\end{enumerate}
